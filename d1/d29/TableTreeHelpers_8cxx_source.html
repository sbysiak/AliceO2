20 auto arrowTypeFromROOT(EDataType type, int size)
21 {
22  auto typeGenerator = [](std::shared_ptr<arrow::DataType>&& type, int size) -> std::shared_ptr<arrow::DataType> {
23  if (size == 1) {
24  return std::move(type);
25  }
26  return arrow::fixed_size_list(type, size);
27  };
28 
29  switch (type) {
30  case EDataType::kBool_t:
31  return typeGenerator(arrow::boolean(), size);
32  case EDataType::kUChar_t:
33  return typeGenerator(arrow::uint8(), size);
34  case EDataType::kUShort_t:
35  return typeGenerator(arrow::uint16(), size);
36  case EDataType::kUInt_t:
37  return typeGenerator(arrow::uint32(), size);
38  case EDataType::kULong64_t:
39  return typeGenerator(arrow::uint64(), size);
40  case EDataType::kChar_t:
41  return typeGenerator(arrow::int8(), size);
42  case EDataType::kShort_t:
43  return typeGenerator(arrow::int16(), size);
44  case EDataType::kInt_t:
45  return typeGenerator(arrow::int32(), size);
46  case EDataType::kLong64_t:
47  return typeGenerator(arrow::int64(), size);
48  case EDataType::kFloat_t:
49  return typeGenerator(arrow::float32(), size);
50  case EDataType::kDouble_t:
51  return typeGenerator(arrow::float64(), size);

116 std::pair<std::shared_ptr<arrow::ChunkedArray>, std::shared_ptr<arrow::Field>> BranchToColumn::read(TBuffer* buffer)
117 {
118  auto totalEntries = static_cast<int>(mBranch->GetEntries());
119  auto status = reserve(totalEntries);
120  if (!status.ok()) {
121  throw runtime_error("Failed to reserve memory for array builder");
122  }
123  int readEntries = 0;
124  buffer->Reset();
125  while (readEntries < totalEntries) {
126  auto readLast = mBranch->GetBulkRead().GetBulkEntries(readEntries, *buffer);
127  readEntries += readLast;
128  status &= appendValues(reinterpret_cast<unsigned char const*>(buffer->GetCurrent()), readLast);
129  }
130  if (!status.ok()) {
131  throw runtime_error("Failed to append values to array");
132  }
133  std::shared_ptr<arrow::Array> array;
134  status &= finish(&array);
135  if (!status.ok()) {
136  throw runtime_error("Failed to create boolean array");
137  }
138  auto fullArray = std::make_shared<arrow::ChunkedArray>(array);
139  auto field = std::make_shared<arrow::Field>(mBranch->GetName(), mArrowType);
140 
141  mBranch->SetStatus(0);
142  mBranch->DropBaskets("all");
143  mBranch->Reset();
144  mBranch->GetTransientBuffer(0)->Expand(0);
145 
146  return std::make_pair(fullArray, field);
147 }

213 ColumnToBranch::ColumnToBranch(TTree* tree, std::shared_ptr<arrow::ChunkedArray> const& column, std::shared_ptr<arrow::Field> const& field)
214  : mBranchName{field->name()},
215  mColumn{column.get()}
216 {
217  auto arrowType = field->type();
218  switch (arrowType->id()) {
219  case arrow::Type::FIXED_SIZE_LIST:
220  mListSize = std::static_pointer_cast<arrow::FixedSizeListType>(arrowType)->list_size();
221  arrowType = arrowType->field(0)->type();
222  break;
223  default:
224  break;
225  }
226  mType = basicROOTTypeFromArrow(arrowType->id());
227  if (mListSize > 1) {
228  mLeafList = mBranchName + "[" + std::to_string(mListSize) + "]" + mType.suffix;
229  } else {
230  mLeafList = mBranchName + mType.suffix;
231  }
232  mBranch = tree->GetBranch(mBranchName.c_str());
233  if (mBranch == nullptr) {
234  mBranch = tree->Branch(mBranchName.c_str(), (char*)nullptr, mLeafList.c_str());
235  }
236  if (mType.type == EDataType::kBool_t) {
237  cache.reserve(mListSize);
238  mCurrent = reinterpret_cast<uint8_t*>(cache.data());
239  mLast = mCurrent + mListSize * mType.size;
240  allocated = true;
241  }
242  accessChunk(0);
243 }

262 void ColumnToBranch::resetBuffer()
263 {
264  if (mType.type == EDataType::kBool_t) {
265  if (O2_BUILTIN_UNLIKELY((*mCurrentPos - mFirstIndex) * mListSize >= getCurrentBuffer()->length())) {
266  nextChunk();
267  }
268  } else {
269  if (O2_BUILTIN_UNLIKELY(mCurrent >= mLast)) {
270  nextChunk();
271  }
272  }
273  accessChunk(*mCurrentPos);
274  mBranch->SetAddress((void*)(mCurrent));

277 void ColumnToBranch::accessChunk(int64_t at)
278 {
279  auto array = getCurrentBuffer();
280 
281  if (mType.type == EDataType::kBool_t) {
282  auto boolArray = std::static_pointer_cast<arrow::BooleanArray>(array);
283  for (auto i = 0; i < mListSize; ++i) {
284  cache[i] = (bool)boolArray->Value((at - mFirstIndex) * mListSize + i);
285  }
286  } else {
287  mCurrent = array->values()->data() + (at - mFirstIndex) * mListSize * mType.size;
288  mLast = mCurrent + array->length() * mListSize * mType.size;
289  }
