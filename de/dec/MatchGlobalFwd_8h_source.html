135  template <typename T>
136  bool propagateToNextClusterWithMCS(T& track, double z, int& startingLayerID, const int& newLayerID)
137  {
138  // Propagate track to the next cluster z position, adding angular MCS effects at the center of
139  // each disk crossed by the track. This method is valid only for track propagation between
140  // clusters at MFT layers positions. The startingLayerID is updated.
141 
142  if (startingLayerID == newLayerID) { // Same layer, nothing to do.
143  LOG(DEBUG) << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
144  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID)
145  << ") ; track.getZ() = " << track.getZ() << " => "
146  << "destination cluster z = " << z << " ; => Same layer: no MCS effects.";
147 
148  if (z != track.getZ()) {
149  track.propagateToZ(z, mBz);
150  }
151  return true;
152  }
153 
154  using o2::mft::constants::LayerZPosition;
155  auto startingZ = track.getZ();
156 
157  //https://stackoverflow.com/questions/1903954/is-there-a-standard-sign-function-signum-sgn-in-c-c
158  auto signum = [](auto a) {
159  return (0 < a) - (a < 0);
160  };
161  int direction = signum(newLayerID - startingLayerID); // takes values +1, 0, -1
162  auto currentLayer = startingLayerID;
163 
164  LOG(DEBUG) << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
165  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID)
166  << ") ; track.getZ() = " << track.getZ() << " => "
167  << "destination cluster z = " << z << " ; ";
168 
169  // Number of disks crossed by this track segment
170  while (currentLayer != newLayerID) {
171  auto nextlayer = currentLayer + direction;
172  auto nextZ = LayerZPosition[nextlayer];
173 
174  int NDisksMS;
175  if (nextZ - track.getZ() > 0) {
176  NDisksMS = (currentLayer % 2 == 0) ? (currentLayer - nextlayer) / 2 : (currentLayer - nextlayer + 1) / 2;
177  } else {
178  NDisksMS = (currentLayer % 2 == 0) ? (nextlayer - currentLayer + 1) / 2 : (nextlayer - currentLayer) / 2;
179  }
180 
181  LOG(DEBUG) << "currentLayer = " << currentLayer << " ; "
182  << "nextlayer = " << nextlayer << " ; "
183  << "track.getZ() = " << track.getZ() << " ; "
184  << "nextZ = " << nextZ << " ; "
185  << "NDisksMS = " << NDisksMS;
186 
187  if ((NDisksMS * mMFTDiskThicknessInX0) != 0) {
188  track.addMCSEffect(NDisksMS * mMFTDiskThicknessInX0);
189  LOG(DEBUG) << "Track covariances after MCS effects:";
190  LOG(DEBUG) << track.getCovariances() << std::endl;
191  }
192 
193  LOG(DEBUG) << " BeforeExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
194 
195  track.propagateToZ(nextZ, mBz);
196 
197  currentLayer = nextlayer;
198  }
199  if (z != track.getZ()) {
200  track.propagateToZ(z, mBz);
201  }
202  startingLayerID = newLayerID;
203  return true;
204  }
