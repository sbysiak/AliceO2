92 std::vector<PNT> EveWorkflowHelper::getTrackPoints(const o2::track::TrackPar& trc, float minR, float maxR, float maxStep)
93 {
94  // adjust minR according to real track start fro track starting point
95  float rMin = std::sqrt(trc.getX() * trc.getX() + trc.getY() * trc.getY());
96  if (rMin > minR) {
97  minR = rMin;
98  }
99  // prepare space points from the track param
100  std::vector<PNT> pnts;
101  int nSteps = std::max(2, int((maxR - minR) / maxStep));
102  const auto prop = o2::base::Propagator::Instance();
103  float xMin = trc.getX(), xMax = maxR * maxR - trc.getY() * trc.getY();
104  if (xMax > 0) {
105  xMax = std::sqrt(xMax);
106  }
107  LOG(INFO) << "R: " << minR << " " << maxR << " || X: " << xMin << " " << xMax;
108  float dx = (xMax - xMin) / nSteps;
109  auto tp = trc;
110  float dxmin = std::abs(xMin - tp.getX()), dxmax = std::abs(xMax - tp.getX());
111  bool res = false;
112  if (dxmin > dxmax) { //start from closest end
113  std::swap(xMin, xMax);
114  dx = -dx;
115  }
116  if (!prop->propagateTo(tp, xMin, false, 0.99, maxStep, o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
117  return pnts;
118  }
119  auto xyz = tp.getXYZGlo();
120  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
121  for (int is = 0; is < nSteps; is++) {
122  if (!prop->propagateTo(tp, tp.getX() + dx, false, 0.99, 999., o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
123  return pnts;
124  }
125  xyz = tp.getXYZGlo();
126  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
127  }
128  return pnts;
129 }
