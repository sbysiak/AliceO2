93 void O2DPLDisplaySpec::run(ProcessingContext& pc)
94 {
95  if (!this->mEveHostNameMatch) {
96  return;
97  }
98 
99  // filtering out any run which occur before reaching next time interval
100  std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
101  std::chrono::duration<double> elapsed = currentTime - this->mTimeStamp;
102  if (elapsed < this->mTimeInteval) {
103  return; // skip this run - it is too often
104  }
105  this->mTimeStamp = currentTime;
106 
107  EveWorkflowHelper helper;
108  helper.getRecoContainer().collectData(pc, *mDataRequest);
109  helper.selectTracks(&(mConfig->configCalib), mClMask, mTrkMask, mTrkMask);
110  helper.draw(this->mJsonPath, this->mNumberOfFiles, this->mNumberOfTracks);
111 }

117 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
118 {
119  WorkflowSpec specs;
120 
121  std::string jsonFolder = cfgc.options().get<std::string>("jsons-folder");
122  std::string eveHostName = cfgc.options().get<std::string>("eve-hostname");
123  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
124  bool useMC = cfgc.options().get<bool>("enable-mc") && !cfgc.options().get<bool>("disable-mc");
125 
126  char hostname[HOST_NAME_MAX];
127  gethostname(hostname, HOST_NAME_MAX);
128  bool eveHostNameMatch = eveHostName.empty() || eveHostName == hostname;
129 
130  int eveDDSColIdx = cfgc.options().get<int>("eve-dds-collection-index");
131  if (eveDDSColIdx != -1) {
132  char* colIdx = getenv("DDS_COLLECTION_INDEX");
133  int myIdx = colIdx ? atoi(colIdx) : -1;
134  if (myIdx == eveDDSColIdx) {
135  LOG(important) << "Restricting DPL Display to collection index, my index " << myIdx << ", enabled " << int(myIdx == eveDDSColIdx);
136  } else {
137  LOG(info) << "Restricting DPL Display to collection index, my index " << myIdx << ", enabled " << int(myIdx == eveDDSColIdx);
138  }
139  eveHostNameMatch &= myIdx == eveDDSColIdx;
140  }
141 
142  std::chrono::milliseconds timeInterval(cfgc.options().get<int>("time-interval"));
143  int numberOfFiles = cfgc.options().get<int>("number-of_files");
144  int numberOfTracks = cfgc.options().get<int>("number-of_tracks");
145 
146  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-tracks"));
147  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-clusters"));
148  if (!srcTrk.any() && !srcCl.any()) {
149  throw std::runtime_error("No input configured");
150  }
151  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
152  dataRequest->requestTracks(srcTrk, useMC);
153  dataRequest->requestClusters(srcCl, useMC);
154 
155  if (cfgc.options().get<bool>("read-from-files")) {
156  InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, useMC);
157  }
158 
159  specs.emplace_back(DataProcessorSpec{
160  "o2-eve-display",
161  dataRequest->inputs,
162  {},
163  AlgorithmSpec{adaptFromTask<O2DPLDisplaySpec>(useMC, srcTrk, srcCl, dataRequest, jsonFolder, timeInterval, numberOfFiles, numberOfTracks, eveHostNameMatch)}});
164 
165  return std::move(specs);
166 }
